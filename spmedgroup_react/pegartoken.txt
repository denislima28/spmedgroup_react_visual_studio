efetuaLogin(event)
    {
      event.preventDefault();

      Axios.post('http://localhost:5000/api/login',
        {
          email : this.state.email,
          senha : this.state.senha
        })
        .then(data => {
          if(data.status === 200){
              console.log(data);
              localStorage.setItem("usuario_spmedgroup", data.data.token);
              //Verifica o tipo de usuário e redireciona para a página default
              console.log(parseJwt().Role);
              if(parseJwt().Role == "ADM"){
                this.props.history.push("/paineladm"); //assim que o login for feito, redireciona para outra página
                console.log(data);
              } else {
                this.props.history.push("/clientes");
              }
              
          } 
      })
      .catch(erro => {
          this.setState({ erroMensagem : 'Email ou senha inválido'});
      })
	}
